# 「开始辩斗」功能完整策划案

## 一、功能概述

**入口**：主菜单 → 点击「开始辩斗」按钮
**目标**：完成一场完整的单人 vs AI 对战
**胜利条件**：先到 8 大势者获胜

**修改议区结算逻辑**：

```typescript
// 主议胜 → +1 大势
// 旁议胜 → 获得1筹（上限1）

function applyLaneRewards(state: DebateBattleState): void {
  // 主议结算
  const zhuYiResult = comparePower(state.player.seats.zhu_yi, state.enemy.seats.zhu_yi);
  if (zhuYiResult.winner === 'player') {
    state.player.resources.daShi += 1;
  } else if (zhuYiResult.winner === 'enemy') {
    state.enemy.resources.daShi += 1;
  }
  // 平局：双方各失1根基（TODO: 平局处理）
  
  // 旁议结算
  const pangYiResult = comparePower(state.player.seats.pang_yi, state.enemy.seats.pang_yi);
  if (pangYiResult.winner === 'player' && state.player.resources.chou < 1) {
    state.player.resources.chou += 1;
  } else if (pangYiResult.winner === 'enemy' && state.enemy.resources.chou < 1) {
    state.enemy.resources.chou += 1;
  }
}

// 比较议区辩锋总和
function comparePower(playerSeat: SeatState, enemySeat: SeatState): { winner: Side | 'draw'; playerPower: number; enemyPower: number } {
  const playerPower = playerSeat.units.reduce((sum, u) => sum + u.power, 0);
  const enemyPower = enemySeat.units.reduce((sum, u) => sum + u.power, 0);
  
  if (playerPower > enemyPower) return { winner: 'player', playerPower, enemyPower };
  if (enemyPower > playerPower) return { winner: 'enemy', playerPower, enemyPower };
  return { winner: 'draw', playerPower, enemyPower };
}
```

## 五、AI策略简化

**Phase 1 最简AI策略**：

```typescript
// src/battleV2/aiStrategy.ts

export function aiSimplePlan(state: DebateBattleState): TurnPlan {
  const hand = state.enemy.hand;
  const budget = state.enemy.resources.cost;
  
  // 1. 随机选一张能打的牌放到主议
  const playableCards = hand.filter(card => card.cost <= budget);
  const mainCard = playableCards[Math.floor(Math.random() * playableCards.length)] || null;
  
  // 2. 如果还有费用，再选一张放到旁议
  const remainingBudget = budget - (mainCard?.cost || 0);
  const secondCard = playableCards.find(card => 
    card !== mainCard && card.cost <= remainingBudget
  ) || null;
  
  return {
    mainCardId: mainCard?.id || null,
    responseCardId: null,
    secretCardId: null,
    writingCardId: null,
    mainTargetSeat: 'zhu_yi',
    secretTargetSeat: 'zhu_yi',
    usedCost: (mainCard?.cost || 0) + (secondCard?.cost || 0),
    lockedPublic: true,
    lockedSecret: true,
  };
}
```

## 六、胜利判定

```typescript
// src/battleV2/engine.ts

function checkVictory(state: DebateBattleState): Side | 'draw' | null {
  if (state.player.resources.daShi >= WIN_DASHI) return 'player';
  if (state.enemy.resources.daShi >= WIN_DASHI) return 'enemy';
  if (state.round >= MAX_ROUNDS) {
    // 超过回合上限，按大势判定
    if (state.player.resources.daShi > state.enemy.resources.daShi) return 'player';
    if (state.enemy.resources.daShi > state.player.resources.daShi) return 'enemy';
    return 'draw';
  }
  return null; // 战斗继续
}
```

```typescript
/**
 * 战斗Reducer - 处理所有战斗动作
 */
export function battleReducer(state: DebateBattleState, action: BattleAction): DebateBattleState {
  switch (action.type) {
    
    // === 计时器Tick ===
    case 'TICK': {
      if (state.phase === 'finished') return state;
      
      const nextSeconds = state.secondsLeft - 1;
      
      // 计时结束，自动推进阶段
      if (nextSeconds <= 0) {
        return advancePhase(state);
      }
      
      return { ...state, secondsLeft: nextSeconds };
    }
    
    // === 选择卡牌放到议区 ===
    case 'PLAN_CARD': {
      const { slot, cardId } = action;
      const player = state.player;
      
      // 验证：只能在明辩/暗谋阶段
      if (state.phase !== 'ming_bian' && state.phase !== 'an_mou') return state;
      
      // 验证：卡牌必须在手牌中
      const card = player.hand.find(c => c.id === cardId);
      if (!card && cardId !== null) return state;
      
      // 验证：费用足够
      if (card && card.cost > player.resources.cost - player.plan.usedCost) {
        console.log('[Engine] 费用不足，无法出牌');
        return state;
      }
      
      // 更新计划
      const nextPlan = { ...player.plan };
      if (slot === 'main') {
        nextPlan.mainCardId = cardId;
        nextPlan.usedCost += card?.cost || 0;
      } else if (slot === 'secret') {
        nextPlan.secretCardId = cardId;
        nextPlan.usedCost += card?.cost || 0;
      }
      
      return {
        ...state,
        player: { ...player, plan: nextPlan },
      };
    }
    
    // === 选择目标议区 ===
    case 'SET_TARGET_SEAT': {
      const { slot, seatId } = action;
      const player = state.player;
      
      const nextPlan = { ...player.plan };
      if (slot === 'main') {
        nextPlan.mainTargetSeat = seatId;
      } else if (slot === 'secret') {
        nextPlan.secretTargetSeat = seatId;
      }
      
      return {
        ...state,
        player: { ...player, plan: nextPlan },
      };
    }
    
    // === 锁定明牌 ===
    case 'LOCK_PUBLIC': {
      if (state.phase !== 'ming_bian') return state;
      
      const player = { ...state.player, plan: { ...state.player.plan, lockedPublic: true } };
      
      // 如果双方都已锁定，提前推进阶段
      const enemyLocked = state.enemy.plan.lockedPublic;
      if (enemyLocked) {
        return advancePhase({ ...state, player });
      }
      
      return { ...state, player };
    }
    
    // === 锁定暗牌 ===
    case 'LOCK_SECRET': {
      if (state.phase !== 'an_mou') return state;
      
      const player = { ...state.player, plan: { ...state.player.plan, lockedSecret: true } };
      
      const enemyLocked = state.enemy.plan.lockedSecret;
      if (enemyLocked) {
        return advancePhase({ ...state, player });
      }
      
      return { ...state, player };
    }
    
    // === AI自动出牌 ===
    case 'AI_AUTO_PLAN': {
      const enemy = state.enemy;
      
      if (state.phase === 'ming_bian' && !enemy.plan.lockedPublic) {
        // AI明辩阶段：选择卡牌放到议区
        const aiPlan = aiPlanMingBian(state);
        const nextEnemy = {
          ...enemy,
          plan: {
            ...aiPlan,
            lockedPublic: true,
          },
        };
        
        // 如果玩家也已锁定，推进阶段
        if (state.player.plan.lockedPublic) {
          return advancePhase({ ...state, enemy: nextEnemy });
        }
        
        return { ...state, enemy: nextEnemy };
      }
      
      if (state.phase === 'an_mou' && !enemy.plan.lockedSecret) {
        // AI暗谋阶段：选择暗牌
        const aiSecret = aiPlanAnMou(state);
        const nextEnemy = {
          ...enemy,
          plan: {
            ...enemy.plan,
            secretCardId: aiSecret.secretCardId,
            secretTargetSeat: aiSecret.secretTargetSeat,
            lockedSecret: true,
          },
        };
        
        if (state.player.plan.lockedSecret) {
          return advancePhase({ ...state, enemy: nextEnemy });
        }
        
        return { ...state, enemy: nextEnemy };
      }
      
      return state;
    }
    
    default:
      return state;
  }
}

/**
 * 阶段推进
 */
function advancePhase(state: DebateBattleState): DebateBattleState {
  const currentPhase = state.phase;
  
  switch (currentPhase) {
    case 'ming_bian':
      // 明辩 → 暗谋
      return {
        ...state,
        phase: 'an_mou',
        secondsLeft: PHASE_DURATION.an_mou,
      };
      
    case 'an_mou':
      // 暗谋 → 揭示
      return {
        ...state,
        phase: 'reveal',
        secondsLeft: PHASE_DURATION.reveal,
      };
      
    case 'reveal':
      // 揭示 → 结算
      return resolvePhase(state);
      
    case 'resolve':
      // 结算 → 回合结束检查
      return endRound(state);
      
    default:
      return state;
  }
}
```

```typescript
/**
 * 结算阶段 - 比较议区、分配奖励、处理伤害
 */
function resolvePhase(state: DebateBattleState): DebateBattleState {
  const logs: BattleLog[] = [];
  const feed: string[] = [];
  
  // === 1. 将卡牌放置到议区 ===
  const playerMainCard = getCardById(state.player, state.player.plan.mainCardId);
  const enemyMainCard = getCardById(state.enemy, state.enemy.plan.mainCardId);
  
  if (playerMainCard) {
    summonUnit(state.player, playerMainCard, state.player.plan.mainTargetSeat);
    feed.push(`我方【${playerMainCard.name}】进入${SEAT_LABEL[state.player.plan.mainTargetSeat]}`);
  }
  if (enemyMainCard) {
    summonUnit(state.enemy, enemyMainCard, state.enemy.plan.mainTargetSeat);
    feed.push(`敌方【${enemyMainCard.name}】进入${SEAT_LABEL[state.enemy.plan.mainTargetSeat]}`);
  }
  
  // === 2. 计算议区控制权 ===
  const laneControls = calculateAllLaneControls(state);
  
  // === 3. 应用议区奖励 ===
  const { playerRewards, enemyRewards } = applyLaneRewards(state, laneControls);
  feed.push(...playerRewards, ...enemyRewards);
  
  // === 4. 单位互攻（主议区双方互打） ===
  const zhuYiDamage = resolveCombat(state.player.seats.zhu_yi, state.enemy.seats.zhu_yi);
  if (zhuYiDamage.playerDamage > 0) {
    applyDamageToUnits(state.enemy.seats.zhu_yi, zhuYiDamage.playerDamage);
    feed.push(`主议：我方对敌方造成 ${zhuYiDamage.playerDamage} 点伤害`);
  }
  if (zhuYiDamage.enemyDamage > 0) {
    applyDamageToUnits(state.player.seats.zhu_yi, zhuYiDamage.enemyDamage);
    feed.push(`主议：敌方对我方造成 ${zhuYiDamage.enemyDamage} 点伤害`);
  }
  
  // === 5. 清理死亡单位 ===
  const deadUnits = removeDeadUnits(state.player.seats.zhu_yi);
  const deadEnemyUnits = removeDeadUnits(state.enemy.seats.zhu_yi);
  if (deadUnits.length > 0) feed.push(`我方 ${deadUnits.map(u => u.name).join('、')} 离场`);
  if (deadEnemyUnits.length > 0) feed.push(`敌方 ${deadEnemyUnits.map(u => u.name).join('、')} 离场`);
  
  // === 6. 检查胜利条件 ===
  const winner = checkVictory(state);
  if (winner) {
    return {
      ...state,
      phase: 'finished',
      winner,
      logs: [...state.logs, ...logs],
      resolveFeed: feed,
    };
  }
  
  return {
    ...state,
    phase: 'resolve',
    secondsLeft: PHASE_DURATION.resolve,
    logs: [...state.logs, ...logs],
    resolveFeed: feed,
  };
}

/**
 * 将卡牌召唤到议区
 */
function summonUnit(player: BattlePlayer, card: DebateCard, seatId: SeatId): boolean {
  const seat = player.seats[seatId];
  
  // 验证：议区未满（最多3张）
  if (seat.units.length >= seat.maxUnits) {
    console.log(`[Engine] ${SEAT_LABEL[seatId]}已满，无法放置`);
    return false;
  }
  
  // 创建单位
  const unit: SeatUnit = {
    id: `unit-${unitSeq++}`,
    cardId: card.id,
    name: card.name,
    power: card.power || 0,
    hp: card.hp || 1,
    maxHp: card.hp || 1,
    position: seat.units.length,
  };
  
  seat.units.push(unit);
  
  // 从手牌移除
  player.hand = player.hand.filter(c => c.id !== card.id);
  
  // 添加到弃牌堆（立论牌在离场时才进弃牌堆，施策牌立即进）
  if (card.type !== '立论') {
    player.discard.push(card);
  }
  
  return true;
}

/**
 * 计算议区伤害交换
 */
function resolveCombat(playerSeat: SeatState, enemySeat: SeatState): {
  playerDamage: number;
  enemyDamage: number;
} {
  // 我方总辩锋
  const playerPower = playerSeat.units.reduce((sum, u) => sum + u.power, 0);
  // 敌方总辩锋
  const enemyPower = enemySeat.units.reduce((sum, u) => sum + u.power, 0);
  
  return {
    playerDamage: playerPower,  // 我方对敌方造成的伤害
    enemyDamage: enemyPower,    // 敌方对我方造成的伤害
  };
}

/**
 * 应用伤害到单位
 */
function applyDamageToUnits(seat: SeatState, totalDamage: number): void {
  if (seat.units.length === 0 || totalDamage <= 0) return;
  
  // 按位置分配伤害（前排先承受）
  let remainingDamage = totalDamage;
  
  for (const unit of seat.units) {
    if (remainingDamage <= 0) break;
    
    // 护印优先吸收
    // TODO: 护印机制
    
    // 扣除根基
    const damageToUnit = Math.min(remainingDamage, unit.hp);
    unit.hp -= damageToUnit;
    remainingDamage -= damageToUnit;
  }
}

/**
 * 移除死亡单位（根基归零）
 */
function removeDeadUnits(seat: SeatState): SeatUnit[] {
  const dead = seat.units.filter(u => u.hp <= 0);
  seat.units = seat.units.filter(u => u.hp > 0);
  return dead;
}

/**
 * 胜利检查
 */
function checkVictory(state: DebateBattleState): Side | 'draw' | null {
  // 大势胜利（先到8）
  if (state.player.resources.daShi >= WIN_DASHI) return 'player';
  if (state.enemy.resources.daShi >= WIN_DASHI) return 'enemy';
  
  // 回合上限兜底
  if (state.round >= MAX_ROUNDS) {
    if (state.player.resources.daShi > state.enemy.resources.daShi) return 'player';
    if (state.enemy.resources.daShi > state.player.resources.daShi) return 'enemy';
    return 'draw';
  }
  
  return null;
}
```

```typescript
/**
 * 回合结束 → 进入下一回合
 */
function endRound(state: DebateBattleState): DebateBattleState {
  const nextRound = state.round + 1;
  const nextMaxCost = getMaxCost(nextRound);
  
  // 重置玩家状态
  const nextPlayer = {
    ...state.player,
    resources: {
      ...state.player.resources,
      cost: nextMaxCost,        // 费用刷新
      maxCost: nextMaxCost,
    },
    plan: {
      mainCardId: null,
      responseCardId: null,
      secretCardId: null,
      writingCardId: null,
      mainTargetSeat: 'zhu_yi',
      secretTargetSeat: 'zhu_yi',
      usedCost: 0,
      lockedPublic: false,
      lockedSecret: false,
    },
  };
  
  const nextEnemy = {
    ...state.enemy,
    resources: {
      ...state.enemy.resources,
      cost: nextMaxCost,
      maxCost: nextMaxCost,
    },
    plan: {
      mainCardId: null,
      responseCardId: null,
      secretCardId: null,
      writingCardId: null,
      mainTargetSeat: 'zhu_yi',
      secretTargetSeat: 'zhu_yi',
      usedCost: 0,
      lockedPublic: false,
      lockedSecret: false,
    },
  };
  
  // 抽牌（手牌不足5张时）
  const playerDrawCount = Math.max(0, 5 - nextPlayer.hand.length);
  const enemyDrawCount = Math.max(0, 5 - nextEnemy.hand.length);
  
  drawCards(nextPlayer, playerDrawCount);
  drawCards(nextEnemy, enemyDrawCount);
  
  return {
    ...state,
    round: nextRound,
    phase: 'ming_bian',
    secondsLeft: PHASE_DURATION.ming_bian,
    player: nextPlayer,
    enemy: nextEnemy,
    resolveFeed: [],
  };
}
```
