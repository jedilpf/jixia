## 二、阶段系统重构

### 2.1 当前阶段流程（旧）

```
回合开始 → 明辩(25秒) → 暗谋(25秒) → 揭示(3秒) → 结算(6秒) → 回合结束
```

**问题**：
- 等待时间过长（总计54秒）
- 明辩/暗谋概念复杂，新手难理解
- 同步博弈需要双方都锁定才能推进

### 2.2 目标阶段流程（新）

```
回合开始 → 出牌(即时) → 结算(即时) → 出牌(即时) → 结算(即时) → 回合结束
```

**改进**：
- 即时反馈，出牌后立刻看到效果
- 简化概念，只理解"出牌→结算"循环
- 单人推进，不需要等待对方锁定

### 2.3 类型定义修改

**文件**: `src/battleV2/types.ts`

### 2.5 状态机逻辑修改

**文件**: `src/battleV2/engine.ts`

```typescript
// 旧流程：需要双方锁定
function advancePhase(state: DebateBattleState): DebateBattleState {
  if (state.phase === 'ming_bian' && player.lockedPublic && enemy.lockedPublic) {
    return { ...state, phase: 'an_mou', secondsLeft: PHASE_DURATION.an_mou };
  }
  // ...
}

// 新流程：单人确认推进
function advancePhase(state: DebateBattleState): DebateBattleState {
  // play_1 → resolve_1：玩家出牌后点击确认
  if (state.phase === 'play_1' && player.plan.mainCardId) {
    return { 
      ...state, 
      phase: 'resolve_1',
      // 立即结算第一张牌
    };
  }
  
  // resolve_1 → play_2：结算动画后自动进入
  if (state.phase === 'resolve_1') {
    return { ...state, phase: 'play_2' };
  }
  
  // play_2 → resolve_2：玩家出第二张牌后确认
  if (state.phase === 'play_2' && player.plan.secondCardId) {
    return { ...state, phase: 'resolve_2' };
  }
  
  // resolve_2 → 回合结束 → play_1（下一回合）
  if (state.phase === 'resolve_2') {
    const nextRound = state.round + 1;
    return {
      ...state,
      round: nextRound,
      phase: 'play_1',
      // 重置费用、抽牌等
    };
  }
}
```

### 3.2 目标胜利条件（新）

```typescript
  cost: number;       // 当前费用
  maxCost: number;    // 本回合费用上限

  dashi: number;      // 大势（到8获胜）
  chou: number;       // 筹（最多1个，减费用）
  guyin: number;      // 护体（护盾）
}

// 移除 hp，新增 dashi、chou、guyin
```

### 3.4 大势获取规则

```typescript
// 主议胜 → +1 大势
function onMainSeatWin(winner: Side): void {
  if (winner === 'player') {
    state.player.resources.dashi += 1;
  } else {
    state.enemy.resources.dashi += 1;
  }
}

// 旁议胜 → +1 筹
function onSideSeatWin(winner: Side): void {
  if (winner === 'player') {
    state.player.resources.chou = 1; // 最多1个
  } else {
    state.enemy.resources.chou = 1;
  }
}

// 胜利检查
function checkVictory(state: DebateBattleState): Side | null {
  if (state.player.resources.dashi >= 8) return 'player';
  if (state.enemy.resources.dashi >= 8) return 'enemy';
  return null;
}
```
